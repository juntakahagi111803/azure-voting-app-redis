
git config --global user.name ""
git config --global user.email ""
git config init.defaultBranch main

git status

#add
git add .\test.txt
git add .

git commit -m "test.txtの1回目保存を実行"
git commit -am "add無しでCommit実行"


git log --oneline -n 2
git log -p

#コミットハッシュのみを選択して確認
git show f65dabe


git diff
#ステージにgit addしたものを確認
git diff --staged


#直近のcommit時のメッセージを修正する場合 --amendを利用する！
git commit --amend -m "04を追加" 


#直近のcommit時の内容を修正する場合(内容を加筆修正したときなど。) --amend --noeditを利用する！これは新しくcommitしない前回のハッシュで良い場合。
git commit --amend --no-edit


#特定のハッシュに戻ったり、戻ったあとまた元に戻ったりするには
git checkout 3973051
git switch -


#特定のハッシュに戻る場合git reset。ただし、もどったら、そのハッシュより新しい各ハッシュは破棄されるので注意！
git reset [--hard --soft --mixed] [Hash]
